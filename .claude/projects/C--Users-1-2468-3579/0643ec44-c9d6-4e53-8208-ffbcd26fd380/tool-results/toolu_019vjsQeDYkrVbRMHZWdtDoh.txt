   686→### 9.2 回放验证逻辑
   687→
   688→```python
   689→class ReplayVerifier:
   690→    """回放验证器 (军规 M7)"""
   691→
   692→    def verify_decision_sequence(
   693→        self,
   694→        original: list[Decision],
   695→        replayed: list[Decision],
   696→    ) -> VerifyResult:
   697→        """验证决策序列一致性"""
   698→        # SHA256哈希比较
   699→        original_hash = self._compute_hash(original)
   700→        replayed_hash = self._compute_hash(replayed)
   701→        return VerifyResult(
   702→            passed=original_hash == replayed_hash,
   703→            original_hash=original_hash,
   704→            replayed_hash=replayed_hash,
   705→        
   706→        )
   707→
   708→    def verify_guardian_sequence(
   709→        self,
   710→        original: list[GuardianAction],
   711→        replayed: list[GuardianAction],
   712→    ) -> VerifyResult:
   713→        """验证守护动作序列一致性"""
   714→
   715→        original_hash = self._compute_hash(original)
   716→        replayed_hash = self._compute_hash(replayed)
   717→        return VerifyResult(
   718→            passed=original_hash == replayed_hash,
   719→            original_hash=original_hash,
   720→            replayed_hash=replayed_hash,
   721→        )
   722→    def _compute_hash(self, sequence: list[Any]) -> str:
   723→        """计算序列的SHA256哈希值"""
   724→        hasher = hashlib.sha256()
   725→        for item in sequence:
   726→            hasher.update(repr(item).encode('utf-8'))
   727→        return hasher.hexdigest() 
   728→```

  1019→class DLModelBase:
  1020→    """深度学习模型基类 (军规 M3, M7, M18)"""

  1031→        """设置确定性随机种子 (军规 M7)"""
  1032→        import random
  1033→        import numpy as np
  1034→        import torch
  1035→
  1036→        random.seed(seed)
  1037→        np.random.seed(seed)
  1038→        torch.manual_seed(seed)
  1039→        if torch.cuda.is_available():
  1040→            torch.cuda.manual_seed_all(seed)
  1041→            torch.backends.cudnn.deterministic = True
  1042→            torch.backends.cudnn.benchmark = False
  1043→
  1044→    def predict(self, inputs: Any) -> tuple[Any, dict]:
  1045→        """预测方法 (必须返回审计信息, 军规 M3)"""
  1046→        raise NotImplementedError
  1047→
  1048→    def can_go_live(self) -> tuple[bool, str]:
  1049→        """检查是否可以上线 (军规 M18)"""
  1050→        from src.strategy.experimental import MaturityEvaluator
  1051→        evaluator = MaturityEvaluator()
  1052→        result = evaluator.evaluate(self)
  1053→        if result.total_score < 0.8:
  1054→            return False, f"成熟度 {result.total_score:.1%} < 80%"

  1063→class RLAgentBase:

  1068→    def __init__(
  1069→        self,

  1075→        self.action_dim = action_dim

  1080→        """选择动作 (必须返回审计信息, 军规 M3)"""
  1081→        raise NotImplementedError
  1082→
  1083→    def update(self, batch: dict) -> dict:
  1084→        """更新模型 (返回训练指标)"""
  1085→        raise NotImplementedError
  1086→
  1087→    def compute_reward(
  1088→        self,
  1089→        pnl: float,
  1090→        position: float,
  1091→        risk_metrics: dict,
  1092→    ) -> float:
  1093→        """计算奖励 (军规 M18, M19)"""
  1094→        # 夏普比率调整奖励
  1095→        sharpe_reward = pnl / (risk_metrics.get("volatility", 1.0) + 1e-8)
  1096→        # 风险惩罚
  1097→        drawdown_penalty = -abs(risk_metrics.get("drawdown", 0)) * 0.5
  1098→        # 持仓成本
  1099→        position_cost = -abs(position) * 0.001
  1100→        return sharpe_reward + drawdown_penalty + position_cost
  1101→```

  1103→#### 11.4.3 交叉验证时序划分器
  1104→
  1105→```python
  1106→class TimeSeriesCVSplitter:
  1107→    """时序交叉验证划分器 (军规 M7)"""
  1108→
  1109→    def __init__(
  1110→        self,
  1111→        n_splits: int = 5,
  1112→        train_size: int | None = None,
  1113→        test_size: int | None = None,
  1114→        gap: int = 0,  # 训练集和测试集之间的间隔
  1115→    ):
  1116→        self.n_splits = n_splits
  1117→        self.train_size = train_size
  1118→        self.test_size = test_size
  1119→        self.gap = gap
  1120→
  1121→    def split(
  1122→        self,
  1123→        X: np.ndarray,
  1124→        y: np.ndarray | None = None,
  1125→    ) -> Generator[tuple[np.ndarray, np.ndarray], None, None]:
  1126→        """生成时序交叉验证划分"""
  1127→        n_samples = len(X)
  1128→        indices = np.arange(n_samples)
  1129→
  1130→        test_size = self.test_size or n_samples // (self.n_splits + 1)
  1131→        train_size = self.train_size or test_size * 2
  1132→
  1133→        for i in range(self.n_splits):
  1134→            test_start = n_samples - (self.n_splits - i) * test_size
  1135→            test_end = test_start + test_size
  1136→            train_end = test_start - self.gap
  1137→            train_start = max(0, train_end - train_size)
  1138→
  1139→            yield indices[train_start:train_end], indices[test_start:test_end]
  1140→```

  1450→
  1451→```python
  1452→class ComplianceThrottle:
  1453→    """合规节流器 (2025年新规)"""
  1454→
  1455→    # 监管阈值
  1456→    REPORT_CANCEL_LIMIT_5S = 50    # 5秒内报撤单上限
  1457→    HIGH_FREQ_LIMIT_PER_SEC = 300  # 高频交易阈值 (笔/秒)
  1458→    HIGH_FREQ_LIMIT_DAILY = 20000  # 高频交易阈值 (笔/日)
  1459→    HIGH_FREQ_LIMIT_PER_MIN = 18000  # 高频交易阈值 (笔/分)
  1460→
  1461→    def __init__(self):
  1462→        self._order_history: deque = deque(maxlen=10000)
  1463→        self._5s_window: deque = deque()
  1464→        self._1min_window: deque = deque()
  1465→        self._daily_count: int = 0
  1466→
  1467→    def can_submit(self) -> tuple[bool, str]:
  1468→        """检查是否可以提交订单"""
  1469→        # 检查5秒窗口
  1470→        now = time.time()
  1471→        while self._5s_window and now - self._5s_window[0] > 5:
  1472→            self._5s_window.popleft()
  1473→
  1474→        # 检查1分钟窗口
  1475→        while self._1min_window and now - self._1min_window[0] > 60:
  1476→            self._1min_window.popleft()
  1477→        if len(self._5s_window) >= self.REPORT_CANCEL_LIMIT_5S:
  1478→            return False, "5秒报撤单频率超限"
  1479→        if len(self._1min_window) >= self.HIGH_FREQ_LIMIT_PER_MIN:
  1480→            return False, "1分钟高频交易频率超限"
  1481→
  1482→        # 检查日内总量
  1483→        # 这里假设有一个方法或机制来更新_daily_count
  1484→        # 例如，每次提交订单时调用 self._daily_count += 1
  1485→        self._5s_window.append(now)        self._1min_window.append(now)
  1486→        self._daily_count += 1
  1487→
  1488→        if self._daily_count >= self.HIGH_FREQ_LIMIT_DAILY:
  1489→            return False, "日内报撤单总量超限"
  1490→
  1491→        return True, ""
  1492→```

  1796→def check_limit_price(
  1797→    order_price: float,
  1798→    instrument: InstrumentInfo,
  1799→    last_settle: float,
  1800→) -> tuple[bool, str]:
  1801→    """检查订单价格是否在涨跌停范围内"""
  1802→
  1803→    limit_up = last_settle * (1 + instrument.price_limit_pct)
  1804→    limit_down = last_settle * (1 - instrument.price_limit_pct)
  1805→
  1806→    if order_price > limit_up:
  1807→        return False, f"订单价格 {order_price} 超过涨停价 {limit_up}"
  1808→
  1809→    if order_price < limit_down:
  1810→        return False, f"订单价格 {order_price} 低于跌停价 {limit_down}"
  1811→
  1812→    return True, ""
  1813→```

  1904→
  1905→```python
  1906→class ChinaFeeCalculator:
  1907→    """中国期货手续费计算器"""
  1908→
  1909→    def calculate(
  1910→        self,
  1911→        instrument: InstrumentInfo,
  1912→        price: float,
  1913→        volume: int,
  1914→        direction: str,  # "OPEN" | "CLOSE" | "CLOSE_TODAY"
  1915→    ) -> float:
  1916→        """计算手续费"""
  1917→
  1918→        if instrument.fee_type == "ratio":
  1919→            # 按金额
  1920→            value = price * instrument.multiplier * volume
  1921→            if direction == "OPEN":
  1922→                return value * instrument.fee_open_ratio
  1923→            elif direction == "CLOSE":
  1924→                return value * instrument.fee_close_ratio
  1925→            else:  # CLOSE_TODAY
  1926→                return value * instrument.fee_close_today_ratio
  1927→
  1928→        elif instrument.fee_type == "fixed":
  1929→            # 按手
  1930→            if direction == "OPEN":
  1931→                return instrument.fee_open_fixed * volume
  1932→            elif direction == "CLOSE":
  1933→                return instrument.fee_close_fixed * volume
  1934→            else:
  1935→                return instrument.fee_close_today_fixed * volume
  1936→
  1937→        else:
  1938→            # 混合: 取两者较高值
  1939→
  1940→            fee_by_value = self.calculate(
  1941→                instrument, price, volume, direction
  1942→            )
  1943→            fee_by_fixed = self.calculate(
  1944→                instrument, price, volume, direction
  1945→            )
  1946→            return max(fee_by_value, fee_by_fixed)
  1947→
  1948→
  1949→```

  2013→## §22 VaR风控增强
  2014→
  2015→### 22.1 当前VaR方法
  2016→
  2017→| 方法 | 优点 | 缺点 | 适用场景 |
  2018→|------|------|------|----------|
  2019→| 历史模拟法 | 简单直观 | 依赖历史 | 一般市场 |
  2020→| 参数法 | 计算快速 | 假设正态 | 快速估计 |
  2021→| 蒙特卡洛法 | 灵活准确 | 计算慢 | 复杂组合 |
  2022→|------|------|------|----------|
  2023→
  2024→### 22.2 增强VaR方法 (待新增)
  2025→
  2026→| 方法 | 描述 | 适用场景 |
  2027→|------|------|----------|
  2028→| EVT (极值理论) | POT + GPD分布 | 极端风险 |
  2029→| 半参数法 | 核密度 + GPD | 厚尾分布 |
  2030→| 涨跌停调整 | 截断效应修正 | 中国期货 |
  2031→| 流动性调整 | 平仓成本建模 | 大额头寸 |
  2032→|------|------|----------|
  2033→
  2034→### 22.3 EVT-GPD实现
  2035→
  2036→```python
  2037→def evt_var(
  2038→    self,
  2039→    returns: list[float],
  2040→    confidence: float = 0.99,
  2041→    threshold_pct: float = 0.95,
  2042→) -> VaRResult:
  2043→    """极值理论VaR (POT + GPD方法)"""
  2044→
  2045→    # 1. 确定阈值 (第95百分位)
  2046→    sorted_returns = sorted(returns)
  2047→    threshold_idx = int(len(sorted_returns) * threshold_pct)
  2048→    threshold = sorted_returns[threshold_idx]
  2049→
  2050→    # 2. 提取超阈值样本
  2051→    exceedances = [r - threshold for r in returns if r < threshold]
  2052→
  2053→    if len(exceedances) < 30:
  2054→        # 样本不足，回退到历史法
  2055→        return self.historical_var(returns, confidence)
  2056→
  2057→    # 3. 估计GPD参数 (矩估计)
  2058→    xi, beta = self._estimate_gpd_params(exceedances)
  2059→
  2060→    # 4. 计算VaR
  2061→    n = len(returns)
  2062→    nu = len(exceedances)
  2063→    p = 1 - confidence
  2064→
  2065→    if xi != 0:
  2066→        var = threshold + (beta / xi) * (((n / nu) * p) ** (-xi) - 1)
  2067→    else:
  2068→        var = threshold - beta * math.log((n / nu) * p)
  2069→
  2070→    return VaRResult(
  2071→        var=abs(var),
  2072→        confidence=confidence,
  2073→        method="EVT-GPD",
  2074→        expected_shortfall=self._evt_es(var, xi, beta),
  2075→        sample_size=len(returns),
  2076→        metadata={"xi": xi, "beta": beta, "threshold": threshold},
  2077→    )
  2078→```
